# Assignment-Triangle-
class Triangel
{
	int height;
	int base;
	int side_a;
	int side_c;
	
	
	Triangle (int h, int b, int c)
	{
		this.height = h;
		this.base = b;
		this.side_ a = a;
		this.side_c = c;
		
	}
	
	int getArea( )
	{
		 return (height * base)/2;
	}
	
	int getPerimeter ( )
	{
		 return side_a + base + side_c;
	}
}
